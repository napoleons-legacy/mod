            primary_culture = yankee

        }

        release_vassal = TEX

        any_owned = {
            limit = {
                OR = {
                    region = MEX_85
                    region = MEX_94
                    region = MEX_97
                    region = MEX_100
                    region = MEX_2138
                    region = MEX_2142
                }
                NOT = { is_core = TEX }

            add_core = CAL

                tag = COL

        diplomatic_influence = { who = LOU value = 400 }
        diplomatic_influence = { who = TEX value = 400 }
