# Greek rebellion starts
country_event = {
    id = 86080
    title = "EVTNAME86080"
    desc = "EVTDESC86080"
    picture = "Riot"

    trigger = {
        tag = TUR
        year = 1839
        has_country_flag = greek_agitation
        NOT = { exists = GRE }
        has_country_flag = greek_revolt
    }

    fire_only_once = yes

    mean_time_to_happen =  {
        months = 3
    }

    option = {
        name = "EVTOPTA86080"

        any_owned = {
            limit = {
                pop_majority_culture = greek
                is_core = GRE
                NOT = {
                    OR = {
                        is_core = CYP
                        is_core = BUL
                        region = TUR_853
                        region = EGY_843
                    }
                }
            }

            secede_province = GRE
        }

        GRE = {
            government = absolute_monarchy
            political_reform = mandatory_service
            capital = 833 # So they don't get that 12% loss due to Athens blockade
            leadership = 80
            treasury = 10000
            small_arms = 100
            ammunition = 200
            canned_food = 200
            wine = 200
            liquor = 200

            any_pop = { militancy = -6 }

            add_country_modifier = {
                name = small_country_draft
                duration = 1825
            }
        }

        set_global_flag = greek_revolt_starts
        war = {
            target = GRE
            attacker_goal = {
                casus_belli = annex_core_country
            }

            defender_goal = {
                casus_belli = status_quo
            }
        }
    }
}
